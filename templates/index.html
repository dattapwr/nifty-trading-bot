import os import requests from flask import Flask, render_template from dhanhq import dhanhq from datetime import datetime app = Flask(__name__) # --- तुमची माहिती भरा --- CLIENT_ID = "1105760761" ACCESS_TOKEN = "तुमचा_टोकन_इथे_टाका" TELEGRAM_BOT_TOKEN = "तुमचा_बॉट_टोकन" TELEGRAM_CHAT_ID = "तुमचा_चॅट_आयडी" dhan = dhanhq(CLIENT_ID, ACCESS_TOKEN) signals = [] def send_telegram(msg): try: url = f"https://api.telegram.org/bot{TELEGRAM_BOT_TOKEN}/sendMessage?chat_id={TELEGRAM_CHAT_ID}&text={msg}" requests.get(url) except: pass @app.route('/') def index(): try: # Crude Oil March Fut (Security ID अंदाजे 420844 असते, एकदा चेक करा) data = dhan.historical_minute_charts(420844, 'MCX', 'INTRA') if data and 'data' in data: candles = data['data'] if len(candles) >= 2: prev = candles[-2] # मागील कँडल curr = candles[-1] # आताची कँडल # अट: लाल कँडल (p_open > p_close) आणि हिरवी कँडल (c_close > c_open) # आणि हिरव्या कँडलची बॉडी लाल कँडलच्या वर क्लोज झाली पाहिजे if prev['open'] > prev['close'] and curr['close'] > curr['open']: if curr['close'] > prev['open']: time_now = datetime.now().strftime("%H:%M:%S") msg = f"🚀 CRUDE OIL BREAKOUT!\nPrice: {curr['close']}\nTime: {time_now}" # डुप्लिकेट मेसेज टाळण्यासाठी तपासणी if not signals or signals[0]['t'] != time_now: send_telegram(msg) signals.insert(0, {'t': time_now, 's': 'CRUDE OIL', 'type': 'BUY', 'p': curr['close']}) return render_template('index.html', stocks=signals[:10], market_open=True) except Exception as e: return f"कनेक्शन एरर: {str(e)}. कृपया तुमचा टोकन आणि इंटरनेट तपासा." if __name__ == "__main__": port = int(os.environ.get("PORT", 10000)) app.run(host='0.0.0.0', port=port)
